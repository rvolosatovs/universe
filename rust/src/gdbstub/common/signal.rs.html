impl Signal {
    #[cfg(not(feature = "paranoid_unsafe"))]
    pub(crate) fn from_protocol_u8(val: u8) -> Signal {
        if val <= 151 {
            // SAFETY: Signal is repr(u8), and `val` was confirmed to fall in valid range
            unsafe { core::mem::transmute(val) }
        } else {
            Signal::UNKNOWN
        }
    }

    #[cfg(feature = "paranoid_unsafe")]
    pub(crate) fn from_protocol_u8(val: u8) -> Signal {
        match val {
            0 => Signal::SIGZERO,
            1 => Signal::SIGHUP,
            2 => Signal::SIGINT,
            3 => Signal::SIGQUIT,
            4 => Signal::SIGILL,
            5 => Signal::SIGTRAP,
            6 => Signal::SIGABRT,
            7 => Signal::SIGEMT,
            8 => Signal::SIGFPE,
            9 => Signal::SIGKILL,
            10 => Signal::SIGBUS,
